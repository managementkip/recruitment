    // --- UI TOGGLES & MODALS ---
    function toggleModal(message = '', isError = false) {
      const modal = document.getElementById('sys-modal');
      const content = document.getElementById('modal-content');
      const icon = document.getElementById('modal-icon');
      
      if (message) {
        content.innerText = message;
        icon.className = isError 
          ? "fa-solid fa-triangle-exclamation text-4xl text-red-600 mb-4 drop-shadow-sm" 
          : "fa-solid fa-circle-check text-4xl text-blue-500 mb-4 drop-shadow-sm";
      }
      modal.classList.toggle('hidden');
    }

    function toggleProcessModal() {
      const modal = document.getElementById('process-modal');

        modal.classList.remove('hidden');

      } else {
        modal.classList.add('opacity-0');

    const getBase64 = (file) => new Promise((resolve, reject) => {
      const reader = new FileReader();
      reader.readAsDataURL(file);
      reader.onload = () => resolve(reader.result.split(',')[1]);
      reader.onerror = error => reject(error);
    });

    // --- FORM SUBMISSIONS ---
    async function submitApplication(e) {
      e.preventDefault();
      const btn = document.getElementById('btn-submit');
      const originalHtml = btn.innerHTML;
      
      btn.innerHTML = `<i class="fa-solid fa-spinner animate-spin"></i> <span class="text-[12px] uppercase font-black tracking-widest">Transmitting...</span>`;
      btn.disabled = true;

      try {
        const fileInput = document.getElementById('app-cv');
        const file = fileInput.files[0];
        if (!file) throw new Error("CV file missing.");

        const base64Data = await getBase64(file);
        
        const payload = {
          name: document.getElementById('app-name').value,
          address: document.getElementById('app-address').value,
          email: document.getElementById('app-email').value,
          wa: document.getElementById('app-wa').value,
          position: document.querySelector('input[name="position"]:checked').value,
          link: document.getElementById('app-link').value,
          cvName: file.name,
          cvMimeType: file.type,
          cvBase64: base64Data
        };

        const response = await server.call('SUBMIT_APPLICATION', payload);
        
        if (response.status === 'success') {
          toggleModal('Data transmitted successfully. Awaiting HR processing.');
          document.getElementById('recruitment-form').reset();
          document.getElementById('cv-filename').classList.add('hidden');
        } else {
          throw new Error(response.message);
        }
      } catch (err) {
        toggleModal('CRITICAL ERR: ' + err.message, true);
      } finally {
        btn.innerHTML = originalHtml;
        btn.disabled = false;
      }
    }

    // --- DASHBOARD DATA FETCHING & RENDERING ---
    async function loadDashboardData() {
      const tbody = document.getElementById('candidate-table-body');
      tbody.innerHTML = `<tr><td colspan="5" class="p-8 text-center text-slate-500"><i class="fa-solid fa-spinner animate-spin mr-2 text-blue-600 text-lg"></i> Loading candidate data...</td></tr>`;

      try {
        const response = await server.call('FETCH_CANDIDATES');
        if (response.status === 'success') {
          allCandidatesData = response.payload; // Store data globally
          applyFilters(); // Automatically render based on current filter dropdowns
        } else {
          throw new Error(response.message);
        }
      } catch (err) {
        tbody.innerHTML = `<tr><td colspan="5" class="p-4 text-center text-red-500 font-bold">Error loading data: ${err.message}</td></tr>`;
      }
    }

    function applyFilters() {
      const posFilter = document.getElementById('filter-position').value;
      const statFilter = document.getElementById('filter-status').value;
      
      let filteredList = allCandidatesData;
      
      // Filter by Position
      if (posFilter !== 'ALL') {
        filteredList = filteredList.filter(c => c.position === posFilter);
      }
      
      // Filter by Status
      if (statFilter !== 'ALL') {
        filteredList = filteredList.filter(c => c.status === statFilter);
      }
      
      renderTable(filteredList);
    }

    function renderTable(candidates) {
      const tbody = document.getElementById('candidate-table-body');
      
      if (candidates.length === 0) {
        tbody.innerHTML = `<tr><td colspan="5" class="p-8 text-center text-slate-500 italic">No candidates found in the database.</td></tr>`;
        return;
      }

      // Map string statuses from Google Sheet to numeric dropdown values
      const statusMap = { 'NEW': 1, 'HR_REVIEW': 2, 'INTERVIEW': 3, 'COGNITIVE': 4, 'TRIAL': 5, 'ONBOARDING': 6, 'ACTIVE': 7 };

      tbody.innerHTML = candidates.map(c => {
        const currentStep = statusMap[c.status] || 1; 
        
        // Format Date and WA Link
        const applyDate = c.date ? new Date(c.date).toLocaleDateString() : 'N/A';
        const cleanWA = c.wa ? c.wa.replace(/\D/g, '') : ''; // Strip non-numbers
        
        return `
          <tr class="border-b border-slate-100 hover:bg-slate-50 transition-colors">
            <td class="p-4 text-xs text-slate-700">
              <div class="font-bold">${c.id}</div>
              <div class="text-[10px] text-slate-400 mt-1"><i class="fa-regular fa-clock"></i> ${applyDate}</div>
            </td>
            <td class="p-4 text-sm text-slate-900">
              <span class="font-bold">${c.name}</span>
              <div class="text-[10px] font-bold mt-1 flex gap-3">
                ${cleanWA ? `<a href="https://wa.me/${cleanWA}" target="_blank" class="text-green-600 hover:text-green-800 transition-colors"><i class="fa-brands fa-whatsapp text-sm"></i> Chat WA</a>` : ''}
              </div>
              <div class="text-[10px] text-slate-500 mt-1 flex gap-3">
                ${c.cvLink ? `<a href="${c.cvLink}" target="_blank" class="hover:text-blue-600 transition-colors"><i class="fa-solid fa-file-pdf"></i> CV</a>` : ''}
                ${c.portfolio ? `<a href="${c.portfolio}" target="_blank" class="hover:text-blue-600 transition-colors"><i class="fa-solid fa-link"></i> Portfolio</a>` : ''}
              </div>
            </td>
            <td class="p-4 text-xs text-slate-600">${c.position}</td>
            <td class="p-4">
              <select onchange="updateCandidateStatus('${c.id}', this.value)" class="text-[10px] font-bold uppercase tracking-widest bg-white border border-slate-300 text-slate-700 py-1.5 px-2 rounded-sm outline-none focus:border-blue-500 focus:ring-1 focus:ring-blue-500 cursor-pointer shadow-sm">
                <option value="1" ${currentStep === 1 ? 'selected' : ''}>01. Submit Form</option>
                <option value="2" ${currentStep === 2 ? 'selected' : ''}>02. HR Review</option>
                <option value="3" ${currentStep === 3 ? 'selected' : ''}>03. Interview</option>
                <option value="4" ${currentStep === 4 ? 'selected' : ''}>04. Cognitive Assessment</option>
                <option value="5" ${currentStep === 5 ? 'selected' : ''}>05. Trial Phase</option>
                <option value="6" ${currentStep === 6 ? 'selected' : ''}>06. Onboarding</option>
                <option value="7" ${currentStep === 7 ? 'selected' : ''}>07. Active Duty</option>
              </select>
            </td>
            <td class="p-4 text-center space-x-3">
              <button onclick="deleteCandidate('${c.id}')" class="text-slate-400 hover:text-red-600 transition-colors cursor-pointer" title="Delete Candidate"><i class="fa-solid fa-trash"></i></button>
            </td>
          </tr>
        `;
      }).join('');
    }

    // --- DASHBOARD ACTIONS (UPDATE & DELETE) ---
    async function updateCandidateStatus(id, stepValue) {
      // Map numeric step back to backend string
      const reverseStatusMap = { '1': 'NEW', '2': 'HR_REVIEW', '3': 'INTERVIEW', '4': 'COGNITIVE', '5': 'TRIAL', '6': 'ONBOARDING', '7': 'ACTIVE' };
      const newStatus = reverseStatusMap[stepValue];

      toggleModal('Updating pipeline and transmitting email...');
      
      try {
        const res = await server.call('UPDATE_STATUS', { id, status: newStatus, adminEmail: currentAdminEmail });
        if (res.status === 'success') {
          toggleModal(res.message);
          loadDashboardData(); // Refresh UI silently in background
        } else throw new Error(res.message);
      } catch (err) {
        toggleModal('CRITICAL ERR: ' + err.message, true);
        loadDashboardData(); // Revert dropdown on failure
      }
    }

    async function deleteCandidate(id) {
      if (!confirm(`Are you sure you want to permanently delete candidate ${id}?`)) return;

      toggleModal('Purging record from database...');
      
      try {
        const res = await server.call('DELETE_CANDIDATE', { id });
        if (res.status === 'success') {
          toggleModal(res.message);
          loadDashboardData(); // Refresh UI
        } else throw new Error(res.message);
      } catch (err) {
        toggleModal('CRITICAL ERR: ' + err.message, true);
      }
    }

    function startCountdownTimer(minutes) {
      let timeInSeconds = minutes * 60;
      const display = document.getElementById('countdown-display');
      
      testTimerInterval = setInterval(() => {
        let m = Math.floor(timeInSeconds / 60);
        let s = timeInSeconds % 60;
        
        display.innerText = `${m < 10 ? '0' : ''}${m}:${s < 10 ? '0' : ''}${s}`;
        
        if (timeInSeconds <= 300) {
          // Under 5 minutes warning
          display.classList.remove('text-white');
          display.classList.add('text-red-500', 'animate-pulse');
        }
        
        if (timeInSeconds <= 0) {
          clearInterval(testTimerInterval);
          toggleModal('Time Limit Reached. Auto-transmitting answers...', true);
          submitAssessmentAnswers(); // Force submit
        }
        timeInSeconds--;
      }, 1000);
    }

    // --- SUBMIT ASSESSMENT & AUTO-GRADE ---
    async function submitAssessmentAnswers() {
      clearInterval(testTimerInterval); // Stop the timer
      const btn = document.getElementById('btn-submit-test');
      const originalHtml = btn.innerHTML;
      
      // 1. Gather all answers dynamically based on how many questions rendered
      const answers = [];
      const totalQuestions = document.querySelectorAll('#questions-render-area > div').length;
      let unanswered = 0;

      for (let i = 0; i < totalQuestions; i++) {
        const selected = document.querySelector(`input[name="q_${i}"]:checked`);
        if (selected) {
          answers.push(selected.value);
        } else {
          answers.push(''); // Blank if they skipped it
          unanswered++;
        }
      }

      // Optional: Prevent submission if questions are skipped
      if (unanswered > 0) {
        toggleModal(`Warning: You have ${unanswered} unanswered question(s). Please complete the matrix.`, true);
        return;
      }

      // 2. Transmit Payload
      btn.innerHTML = `<i class="fa-solid fa-spinner animate-spin"></i> <span class="text-[12px] uppercase font-black tracking-widest">Processing Matrix...</span>`;
      btn.disabled = true;

      try {
        const payload = {
          token: currentCandidateToken,
          answers: answers
        };

        const response = await server.call('SUBMIT_ASSESSMENT', payload);
        
        if (response.status === 'success') {
          // Hide the test and show a completion message
          document.getElementById('questions-render-area').innerHTML = `
            <div class="bg-white border border-slate-200 p-12 rounded shadow-sm w-full text-center">
              <i class="fa-solid fa-check-circle text-5xl text-blue-500 mb-4"></i>
              <h3 class="text-2xl font-black tracking-tight text-slate-800 mb-2">Transmission Complete</h3>
              <p class="text-sm text-slate-500">Your cognitive assessment has been logged. HR will review your results and contact you shortly.</p>
            </div>
          `;
        } else {
          throw new Error(response.message);
        }
      } catch (err) {
        toggleModal('SYS ERR: ' + err.message, true);
        btn.innerHTML = originalHtml;
        btn.disabled = false;
      }
    }

              : `<span class="bg-orange-100 text-orange-700 border border-orange-200 px-2 py-1 rounded-sm text-[10px] font-bold tracking-widest uppercase"><i class="fa-solid fa-clock mr-1"></i> Pending Signature</span>`;
            
            const linkButton = hasSigned
              ? `<a href="${c.pdfUrl}" target="_blank" class="bg-red-50 text-red-600 hover:bg-red-100 hover:text-red-700 border border-red-200 px-3 py-1.5 rounded-sm text-[10px] font-bold tracking-widest uppercase transition-colors inline-flex items-center gap-2"><i class="fa-solid fa-file-pdf"></i> View PDF</a>`
              : `<span class="text-slate-400 text-[10px] uppercase font-bold tracking-widest">Unavailable</span>`;

            return `
              <tr class="border-b border-slate-100 hover:bg-slate-50 transition-colors">
                <td class="p-4"><div class="font-bold text-sm text-slate-900">${c.name}</div><div class="text-[10px] text-slate-400 mt-1 font-mono">${c.id}</div></td>
                <td class="p-4 text-xs text-slate-600 font-bold">${c.position}</td>
                <td class="p-4">${statusBadge}</td>
                <td class="p-4 text-center">${linkButton}</td>
              </tr>
            `;
          }).join('');
        } else throw new Error(response.message);
      } catch (err) {
        tbody.innerHTML = `<tr><td colspan="4" class="p-4 text-center text-red-500 font-bold">Error: ${err.message}</td></tr>`;
      }
    }

    // --- SCORE FETCHING & RUBRIC LOGIC ---
    async function loadScoresData() {
      const tbody = document.getElementById('score-table-body');
      tbody.innerHTML = `<tr><td colspan="4" class="p-8 text-center text-slate-500"><i class="fa-solid fa-spinner animate-spin mr-2 text-blue-600 text-lg"></i> Compiling Analytics...</td></tr>`;

      try {
        const response = await server.call('FETCH_SCORES');
        if (response.status === 'success') {
          renderScoresTable(response.payload);
        } else throw new Error(response.message);
      } catch (err) {
        tbody.innerHTML = `<tr><td colspan="4" class="p-4 text-center text-red-500 font-bold">Error: ${err.message}</td></tr>`;
      }
    }

    function interpretScore(score) {
      if (score >= 180) return { cat: 'Sangat Cocok', rec: 'Sangat direkomendasikan', color: 'text-green-700 bg-green-50 border-green-200' };
      if (score >= 165) return { cat: 'Cocok Baik', rec: 'Direkomendasikan, interview lebih dalam', color: 'text-emerald-700 bg-emerald-50 border-emerald-200' };
      if (score >= 150) return { cat: 'Cukup', rec: 'Bisa dipertimbangkan', color: 'text-blue-700 bg-blue-50 border-blue-200' };
      if (score >= 130) return { cat: 'Kurang Cocok', rec: 'Risiko tinggi, evaluasi ketat', color: 'text-orange-700 bg-orange-50 border-orange-200' };
      return { cat: 'Tidak Direkomendasikan', rec: 'Hindari untuk posisi ini', color: 'text-red-700 bg-red-50 border-red-200' };
    }

    function renderScoresTable(scores) {
      const tbody = document.getElementById('score-table-body');
      if (scores.length === 0) {
        tbody.innerHTML = `<tr><td colspan="4" class="p-8 text-center text-slate-500 italic">No assessment data found.</td></tr>`;
        return;
      }

      tbody.innerHTML = scores.map(s => {
        const interpretation = interpretScore(s.score);
        return `
          <tr class="border-b border-slate-100 hover:bg-slate-50 transition-colors">
            <td class="p-4">
              <div class="font-bold text-sm text-slate-900">${s.name}</div>
              <div class="text-[10px] text-slate-400 mt-1 font-mono">${s.id}</div>
            </td>
            <td class="p-4 text-xs text-slate-600 font-bold">${s.position}</td>
            <td class="p-4 text-center">
              <span class="text-lg font-black font-mono text-slate-800">${s.score}</span>
            </td>
            <td class="p-4">
              <div class="inline-block px-3 py-2 border rounded-sm ${interpretation.color} shadow-sm w-full max-w-xs">
                <div class="font-black tracking-tighter uppercase italic text-xs mb-1 drop-shadow-sm">${interpretation.cat}</div>
                <div class="text-[10px] uppercase font-bold tracking-widest opacity-80 leading-tight">${interpretation.rec}</div>
              </div>
            </td>
          </tr>
        `;
      }).join('');
    }

    // --- ADMIN PREVIEW ASSESSMENT LOGIC ---
    let currentPreviewData = []; // Store matrix data globally for editing
    let currentPreviewRole = ''; // Store the role being edited

    async function previewAdminAssessment(role) {
      currentPreviewRole = role;
      const renderArea = document.getElementById('admin-preview-render');
      renderArea.classList.remove('hidden');
      renderArea.classList.add('flex');
      renderArea.innerHTML = `<div class="p-8 text-center text-slate-500"><i class="fa-solid fa-spinner animate-spin mr-2 text-blue-600 text-lg"></i> Fetching ${role} matrix from database...</div>`;

      try {
        const response = await server.call('FETCH_PREVIEW', { role: role });
        
        if (response.status === 'success') {
          currentPreviewData = response.payload;
          let htmlBlock = `<h3 class="font-black tracking-tighter uppercase italic text-md text-slate-800 mb-2 mt-4">Previewing: ${role} Questions</h3>`;
          
          currentPreviewData.forEach((q, index) => {
            const qNum = q.num || (index + 1); 
            htmlBlock += `
              <div class="bg-white border border-slate-200 p-4 rounded shadow-sm w-full relative group">
                <button onclick="openEditQModal(${index})" class="absolute top-4 right-4 text-slate-300 hover:text-blue-600 transition-colors cursor-pointer" title="Edit Question">
                  <i class="fa-solid fa-pen"></i>
                </button>
                <h4 class="font-bold text-sm text-slate-800 mb-3 pr-8"><span class="text-blue-600 mr-2">${qNum}.</span> ${q.text}</h4>
                <div class="grid grid-cols-2 gap-2 pl-4">
                  <div class="text-[11px] text-slate-600 bg-slate-50 p-2 rounded border border-slate-100 flex justify-between"><span><strong>A.</strong> ${q.optA}</span> <span class="text-blue-400 font-bold">[${q.scoreA}]</span></div>
                  <div class="text-[11px] text-slate-600 bg-slate-50 p-2 rounded border border-slate-100 flex justify-between"><span><strong>B.</strong> ${q.optB}</span> <span class="text-blue-400 font-bold">[${q.scoreB}]</span></div>
                  <div class="text-[11px] text-slate-600 bg-slate-50 p-2 rounded border border-slate-100 flex justify-between"><span><strong>C.</strong> ${q.optC}</span> <span class="text-blue-400 font-bold">[${q.scoreC}]</span></div>
                  <div class="text-[11px] text-slate-600 bg-slate-50 p-2 rounded border border-slate-100 flex justify-between"><span><strong>D.</strong> ${q.optD}</span> <span class="text-blue-400 font-bold">[${q.scoreD}]</span></div>
                </div>
              </div>
            `;
          });
          renderArea.innerHTML = htmlBlock;
        } else {
          throw new Error(response.message);
        }
      } catch (err) {
        renderArea.innerHTML = `<div class="p-4 text-center text-red-500 font-bold">Error loading matrix: ${err.message}</div>`;
      }
    }

    // --- EDIT QUESTION LOGIC ---
    function toggleEditQModal() {
      const modal = document.getElementById('edit-q-modal');
      if (modal.classList.contains('hidden')) {
        modal.classList.remove('hidden');
        void modal.offsetWidth; 
        modal.classList.remove('opacity-0');
      } else {
        modal.classList.add('opacity-0');
        setTimeout(() => modal.classList.add('hidden'), 300); 
      }
    }

    function openEditQModal(index) {
      const q = currentPreviewData[index];
      document.getElementById('edit-q-role-label').innerText = `Role: ${currentPreviewRole} | Q-ID: ${q.num}`;
      document.getElementById('edit-q-row').value = q.rowIndex;
      document.getElementById('edit-q-text').value = q.text;
      document.getElementById('edit-q-optA').value = q.optA;
      document.getElementById('edit-q-scoreA').value = q.scoreA;
      document.getElementById('edit-q-optB').value = q.optB;
      document.getElementById('edit-q-scoreB').value = q.scoreB;
      document.getElementById('edit-q-optC').value = q.optC;
      document.getElementById('edit-q-scoreC').value = q.scoreC;
      document.getElementById('edit-q-optD').value = q.optD;
      document.getElementById('edit-q-scoreD').value = q.scoreD;
      
      toggleEditQModal();
    }

    async function saveQuestionEdit(e) {
      e.preventDefault();
      const btn = document.getElementById('btn-save-q');
      const originalHtml = btn.innerHTML;
      btn.innerHTML = `<i class="fa-solid fa-spinner animate-spin"></i> Committing...`;
      btn.disabled = true;

      try {
        const payload = {
          role: currentPreviewRole,
          rowIndex: document.getElementById('edit-q-row').value,
          text: document.getElementById('edit-q-text').value,
          optA: document.getElementById('edit-q-optA').value,
          scoreA: document.getElementById('edit-q-scoreA').value,
          optB: document.getElementById('edit-q-optB').value,
          scoreB: document.getElementById('edit-q-scoreB').value,
          optC: document.getElementById('edit-q-optC').value,
          scoreC: document.getElementById('edit-q-scoreC').value,
          optD: document.getElementById('edit-q-optD').value,
          scoreD: document.getElementById('edit-q-scoreD').value,
        };

        const response = await server.call('UPDATE_QUESTION', payload);
        
        if (response.status === 'success') {
          toggleEditQModal();
          toggleModal(response.message);
          previewAdminAssessment(currentPreviewRole); // Refresh the preview list
        } else throw new Error(response.message);
      } catch (err) {
        toggleModal('SYS ERR: ' + err.message, true);
      } finally {
        btn.innerHTML = originalHtml;
        btn.disabled = false;
      }
    }
